Core Data Models (Pydantic)

classification/models.py

from pydantic import BaseModel, Field from typing import Optional, List from datetime import datetime from enum import Enum class ClassificationMethod(str, Enum): MANUAL_OVERRIDE = "manual_override" EXACT_HISTORICAL = "exact_historical_match" FUZZY_SUPPLIER = "fuzzy_supplier_match" COMMODITY_MATCH = "commodity_match" DESCRIPTION_MATCH = "description_match" HISTORICAL_PATTERN = "historical_pattern" NON_IT_SUGGESTION = "non_it_vendor_suggestion" NO_MATCH = "no_match" class InvoiceLine(BaseModel): """Invoice line from Coupa""" invoice_id: str line_number: int invoice_date: datetime supplier: str description: Optional[str] = None commodity: Optional[str] = None total_amount: float currency: str it_activity_code: Optional[str] = None # Existing code (if any) class ActivityCode(BaseModel): """Activity code from Activity Codes_Active sheet""" activity_code: str activity_description: str it_activity_code: str level1_type: str level2_category: str level3_name: str activity_definition: Optional[str] = None supplier_names: Optional[str] = None # Pipe-delimited class ClassificationSuggestion(BaseModel): """Single classification suggestion""" activity_code: str confidence: float = Field(ge=0, le=100) rationale: str class ClassificationResult(BaseModel): """Result of classification attempt""" activity_code: Optional[str] confidence: float = Field(ge=0, le=100) method: ClassificationMethod needs_review: bool admin_alert: bool = False suggestions: Optional[List[ClassificationSuggestion]] = None reason: Optional[str] = None classified_at: datetime = Field(default_factory=datetime.utcnow) class AdminAlert(BaseModel): """Admin alert for non-IT vendor or ambiguous case""" classification_history_id: int alert_type: str # non_it_vendor, high_value, ambiguous alert_message: str resolved: bool = False created_at: datetime = Field(default_factory=datetime.utcnow)

Key Model Validations

  • confidence: 0-100 range enforced via Pydantic Field(ge=0, le=100)
  • activity_code: Validated against activity_codes table (database constraint)
  • supplier: Normalized (uppercase, trimmed) before matching
  • invoice_date: ISO 8601 datetime parsing

Classification Engine API

classification/engine.py

class ClassificationEngine: """Main classification engine""" def __init__(self, db_session): self.db = db_session self.fuzzy_matcher = FuzzyMatcher(db_session) self.keyword_matcher = KeywordMatcher(db_session) self.historical_matcher = HistoricalMatcher(db_session) self.non_it_detector = NonITDetector() def classify(self, invoice_line: InvoiceLine) -> ClassificationResult: """ Classify a single invoice line. Returns ClassificationResult with: - activity_code (or None if no match) - confidence score (0-100) - classification method - needs_review flag - admin_alert flag (for non-IT vendors) - suggestions (for low-confidence cases) """ # Stage 1: Check manual overrides (100% confidence) if override := self._check_manual_override(invoice_line): return override # Stage 2: Exact historical match (95% confidence) if exact := self.historical_matcher.exact_match(invoice_line.supplier): return exact # Stage 3: Fuzzy supplier match (60-90% confidence) if fuzzy := self.fuzzy_matcher.match(invoice_line.supplier): return fuzzy # Stage 4: Commodity match (50-80% confidence) if invoice_line.commodity: if commodity := self.keyword_matcher.match_commodity(invoice_line.commodity): return commodity # Stage 5: Description match (40-70% confidence) if invoice_line.description: if desc := self.keyword_matcher.match_description(invoice_line.description): return desc # Stage 6: Historical pattern (50-75% confidence) if pattern := self.historical_matcher.pattern_match(invoice_line.supplier): return pattern # Stage 7: Non-IT vendor detection if self.non_it_detector.is_non_it(invoice_line.supplier): suggested_code = self.non_it_detector.suggest_code(invoice_line) return ClassificationResult( activity_code=suggested_code, confidence=30, method=ClassificationMethod.NON_IT_SUGGESTION, needs_review=True, admin_alert=True, reason=f"Non-IT vendor: {self.non_it_detector.get_matched_keyword(invoice_line.supplier)}" ) # Stage 8: No match found return ClassificationResult( activity_code=None, confidence=0, method=ClassificationMethod.NO_MATCH, needs_review=True, admin_alert=False, reason="New supplier with no historical data or keyword matches" ) def batch_classify(self, invoice_lines: List[InvoiceLine]) -> List[ClassificationResult]: """Classify multiple invoices efficiently""" results = [] for line in invoice_lines: result = self.classify(line) self._store_result(line, result) results.append(result) return results

MCP Tool Integration

src/coupa_mcp/tools/classification.py

from mcp.server import tool from classification.engine import ClassificationEngine from classification.models import InvoiceLine @tool() async def coupa_classify_invoice( invoice_id: str, line_number: int ) -> dict: """ Classify a single invoice line with activity code. Returns classification result with confidence score and rationale. """ # Fetch invoice from Coupa API invoice = await coupa_client.get_invoice(invoice_id) line = invoice.lines[line_number - 1] # Convert to InvoiceLine model invoice_line = InvoiceLine( invoice_id=invoice_id, line_number=line_number, supplier=line.supplier, # ... other fields ) # Classify engine = ClassificationEngine(db_session) result = engine.classify(invoice_line) return result.model_dump() @tool() async def coupa_batch_classify_invoices( start_date: str, end_date: str, confidence_threshold: float = 70.0 ) -> dict: """ Batch classify invoices in date range. Returns summary of auto-assigned, needs review, and admin alerts. """ # Fetch invoices invoices = await coupa_client.list_invoices(start_date, end_date) # Extract lines invoice_lines = extract_lines(invoices) # Classify engine = ClassificationEngine(db_session) results = engine.batch_classify(invoice_lines) # Summarize summary = { "total_lines": len(results), "auto_assigned": sum(1 for r in results if r.confidence >= confidence_threshold), "needs_review": sum(1 for r in results if r.needs_review), "admin_alerts": sum(1 for r in results if r.admin_alert) } return summary @tool() async def coupa_get_admin_alerts(status: str = "pending") -> dict: """Get admin alerts for non-IT vendors requiring approval.""" alerts = db_session.query(AdminAlertModel).filter_by( resolved=(status == "resolved") ).all() return {"alerts": [alert.to_dict() for alert in alerts]}

Performance Considerations

Optimization Strategies

  • Batch Processing: Process 1000+ invoices efficiently using pandas DataFrames
  • Fuzzy Matching Cache: Cache RapidFuzz results for repeated supplier names
  • Database Indexes: Index supplier_name, invoice_id for fast lookups
  • Lazy Loading: Load activity codes once at startup, not per classification
  • Parallel Processing: Use multiprocessing for >10K invoice batches
